## 分析

```js
const oh = (!~+[] + {})[--[~+''][+[]] * [~+[]] + ~~!+[]] + ({} + [])[[~!+[]] * ~+[]]

// (!~+[] + {})
// +[]          -> 0                      Number([]) 结果 0
// ~0           -> -1                     按位非
// !-1          -> false                  -1 是真值，取反
// false + {}   -> false[object Object]   先调用 ({}).toString() 方法得 [object Object] 然后字符串拼接

// [--[~+''][+[]] * [~+[]] + ~~!+[]]
// +'' -> 0
// ~0  -> -1
// --[-1]
```

## 会有什么效果

```js
/* No.1 */
function foo() {
  foo()
}
foo()

/* No.2 */
function foo() {
  setTimeout(foo, 0)
}
foo()

/* No.3 */
function foo() {
  return Promise.resolve().then(foo)
}
foo()

/* No.4 */
async function foo() {
  return await foo()
}
foo()
```
